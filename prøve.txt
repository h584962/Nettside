        String switchStatement(int a) {
            String response = "";
            switch(a) {
                case 0:
                    response = "null";
                    break;
                case 1:
                    response = "en";
                    break;
                case 2:
                    response = "to";
                    break;
                case 3:
                    response = "tre";
                    break;
                case 4:
                    response = "fire";
                    break;
                default:
                    response = "ulovlig verdi";
            }
            return response;
        }
